# Types

## `FactoryGenerator`

```ts
type FactoryGenerator = <T>(fake: Faker.FakerStatic) => IDataObject<T>;
```

> For more information on Faker.FakerStatic type refer to https://www.npmjs.com/package/@types/faker

## `IFactoryObject`

```ts
interface IFactoryObject<T = any> {
  make: StateGenerator<T>;
  create: PromiseGenerator<T>;
  only: (keys: keyof T | Array<keyof T>, overrides?: IDataObject) => Partial<T>;
  seed: (value: number) => IFactoryObject<T>;
  state: (name: string, stateValues: IDataObject) => void;
  configDatabase(options: DatabaseConfig<T>): void;
}
```

## `StateGenerator`

```ts
interface StateGenerator<T> {
  (): T;
  (overrides: IDataObject): T;
  (count: number): T[];
  (count: number, overrides: IDataObject): T[];
}
```

## `PromiseGenerator`

```ts
interface PromiseGenerator<T> {

}
```
